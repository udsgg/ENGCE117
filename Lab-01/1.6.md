```c++
#include <stdio.h>
    
int *GetMatrix(int *row, int *col) {
    static int data[100];
    int r, c;
    int index = 0;

    printf("Enter number of rows:");
    scanf("%d", row);
    printf("Enter number of columns:");
    scanf("%d", col);
    printf("Enter matrix values:\n");
    for (r = 0; r < *row; r++) {
        for (c = 0; c < *col; c++) {
            printf("Enter member value [%d][%d]:", r + 1, c + 1);
            scanf("%d", &data[index]);
            index++;
        }
    }
    return data;
}

int main() {
    int *data, m, n;
    data = GetMatrix(&m, &n);
    return 0;
}
```
