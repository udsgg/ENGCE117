int GetSet( int arr[], int size );

        return 1;
    }
    int arr[num]; 
    GetSet( arr, num );
    for (int i = 0; i < num; i++) {
        printf("Member %d: %d\n", i + 1, arr[i]);
    }
    
    return 0;
}

    for (int i = 0; i < size; i++) {

}
