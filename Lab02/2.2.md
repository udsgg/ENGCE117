```c++
#include <stdio.h>
#include <string.h>

char* reverse(char str1[]);

int main() {
    char text[50] = "I Love You";
    char *out;

    out = reverse(text);
    printf("%s\n", out);

    return 0;
}

char* reverse(char str1[]) {
    static char result[50];   // ต้องใช้ static เพื่อให้ใช้ได้หลัง return
    int len = strlen(str1);
    int i;

    for(i = 0; i < len; i++) {
        result[i] = str1[len - 1 - i];
    }

    result[len] = '\0';
    return result;
}
```
