void explode(char str1[], char splitter, char str2[][10], int *count) {
    int i = 0, j = 0, k = 0;
    *count = 0;
    while (str1[i] != '\0') {
        if (str1[i] == splitter) {
            str2[k][j] = '\0';  
            k++;
            j = 0;
        } else {
            str2[k][j] = str1[i];
            j++;
        }
        i++;
    }
    str2[k][j] = '\0';  
    *count = k + 1;
}
