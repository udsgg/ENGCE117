#include <stdio.h>
#include <string.h>

struct studentNode {
    char name[ 20 ] ;
    int age ;
    char sex ;
    float gpa ;
    struct studentNode *next ;
} ;

class linkedList {
    protected :
        struct studentNode *start, **now ;
    public :
        linkedList() ;
        ~linkedList() ;
        void InsNode( const char n[], int a, char s, float g ) ;
        void DelNode() ;
        void GoNext() ;
        void Gofirst() ;
        void GoLast() ;
        void ShowAll() ;
        int FindNode( const char n[] ) ;
        struct studentNode *NowNode() ;
        void EditNode( const char n[], int a, char s, float g ) ;  
} ;

linkedList::linkedList() {
    start = NULL ;
    now = &start ;
}

linkedList::~linkedList() {
    struct studentNode *temp = start;
    while(temp != NULL) {
        struct studentNode *nextNode = temp->next;
        delete temp;
        temp = nextNode;
    }
}

void EditData (linkedList *ll) ;

        scanf ( "%d", &menu ) ;
    }
    
    writefile( &listA ) ; 
    return 0;
}

void readfile (linkedList *ll) {
    FILE *fp ;
    fp = fopen ( "D:\\engcee\\ABC.dat" , "r");
    if ( fp == NULL ) {
        printf ( "File not found\n" ) ;
        return ;
    }
    char name[ 20 ] ;
    int age ;
    char sex ;
    float gpa ;
    
    while ( fscanf ( fp, "%s %d %c %f", name, &age, &sex, &gpa ) != EOF ) {
        ll->InsNode( name, age, sex, gpa ) ;

void AddData( linkedList *ll ) {
    char n[ 20 ] ; int a ; char s ; float g ;
    printf(" Enter Name Age Sex GPA (ex. John 20 M 3.50): ") ;
    scanf("%s %d %c %f", n, &a, &s, &g) ;
    ll->InsNode(n, a, s, g) ;
    printf(" Added Successfully!\n") ;
}

void EditData( linkedList *ll ) {
    char target[ 20 ] ;
    printf(" Enter name to edit: ") ;
    scanf("%s", target) ;
    
    if(ll->FindNode(target) == 1) {
        int a ; char s ; float g ;
        printf(" Found! Enter NEW Age Sex GPA: ") ;
        scanf("%d %c %f", &a, &s, &g) ;
        ll->EditNode(target, a, s, g) ;
        printf(" Edit Successful!\n") ;
    } else {
        printf(" Name not found!\n") ;
    }
}

void FindData( linkedList *ll ) {
    char target[ 20 ] ;
    printf(" Enter name to search: ") ;
    scanf("%s", target) ;
    
    if(ll->FindNode(target) == 1) {
        struct studentNode *node = ll->NowNode() ;
        printf(" Found -> Name: %s, Age: %d, Sex: %c, GPA: %.2f\n", node->name, node->age, node->sex, node->gpa) ;
    } else {
        printf(" Name not found!\n") ;
    }
}

void writefile( linkedList *ll ) {
    FILE *fp ;
    fp = fopen("D:\\engcee\\ABC.dat", "w") ;
    if(fp == NULL) {
        printf(" Error: Cannot save to file.\n") ;
        return ;
    }
    
    ll->Gofirst() ;
    struct studentNode *node = ll->NowNode() ;
    
    while(node != NULL) {
        fprintf(fp, "%s %d %c %.2f\n", node->name, node->age, node->sex, node->gpa) ;
        ll->GoNext() ;
        node = ll->NowNode() ;
    }
    
    fclose(fp) ;
    printf("Data saved to file successfully.\n") ;
}

void linkedList::Gofirst() {
    now = &start ;
}

void linkedList::GoNext() {
    if(now != NULL && *now != NULL) {
        now = &((*now)->next) ;
    }
}

void linkedList::GoLast() {
    now = &start ;
    while(*now != NULL && (*now)->next != NULL) {
        now = &((*now)->next) ;
    }
}

struct studentNode* linkedList::NowNode() {
    if(now != NULL && *now != NULL) {
        return *now ;
    }
    return NULL ;
}

void linkedList::InsNode( const char n[], int a, char s, float g ) {
    now = &start ;
    while(*now != NULL) {
        now = &((*now)->next) ;
    }
    
    *now = new struct studentNode ;
    strcpy((*now)->name, n) ;
    (*now)->age = a ;
    (*now)->sex = s ;
    (*now)->gpa = g ;
    (*now)->next = NULL ;
}

int linkedList::FindNode( const char n[] ) {
    now = &start ;
    while(*now != NULL) {
        if(strcmp((*now)->name, n) == 0) {
            return 1 ;
        }
        now = &((*now)->next) ;
    }
    return 0 ;
}

void linkedList::DelNode() {
    char target[ 20 ] ;
    printf(" Enter name to delete: ") ;
    scanf("%s", target) ;
    
    if(FindNode(target) == 1) {
        struct studentNode *temp = *now ;
        *now = temp->next ;
        delete temp ;
        printf("Delete Successful!\n") ;
    } else {
        printf("Name not found!\n") ;
    }
}

void linkedList::EditNode( const char n[], int a, char s, float g ) {
    if(now != NULL && *now != NULL) {
        strcpy((*now)->name, n) ;
        (*now)->age = a ;
        (*now)->sex = s ;
        (*now)->gpa = g ;
    }
}

void linkedList::ShowAll() {
    struct studentNode *walk = start ;
    printf("\n --- Student List ---\n") ;
    if(walk == NULL) printf(" No data available.\n") ;
    while(walk != NULL) {
        printf(" %-10s Age: %-2d Sex: %-2c GPA: %.2f\n", walk->name, walk->age, walk->sex, walk->gpa) ;
        walk = walk->next ;
    }
    printf(" --------------------\n") ;
}
