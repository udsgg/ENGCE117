#include <stdio.h>

struct student {
    char name[20];
    int age;
};

struct student (*GetStudent(int *room))[10];

int main() {
    struct student (*children)[10];
    int group;
    children = GetStudent(&group);
    printf("\n--- Class info ---\n");
    for (int i = 0; i < 10; i++) {
        printf("Student %d: %s\n", i + 1, (*children)[i].name);
    }
    return 0;
}

struct student (*GetStudent(int *room))[10] {
    static struct student class_room[10];  
    *room = 1;
    for (int i = 0; i < 10; i++) {
        printf("Student %d Name: ", i + 1);
        scanf("%s", class_room[i].name);
    }
    return &class_room;
}
