```C
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct studentNode {
    char name[20];
    int age;
    char sex;
    float gpa;
    struct studentNode *next;
    struct studentNode *back;
};

void ShowAll(struct studentNode *walk);
struct studentNode *addnode(struct studentNode **start, char *n, int a, char s, float g);
void InsNode(struct studentNode *now, char *n, int a, char s, float g);
void Goback(struct studentNode **now);
void DelNode(struct studentNode **now, struct studentNode **start); // ต้องรับ **start ด้วย

int main() {
    struct studentNode *start, *now;
    start = NULL;
    
    // 1. Add Nodes
    now = addnode(&start, "one", 6, 'M', 3.11); ShowAll(start);
    now = addnode(&start, "two", 8, 'F', 3.22); ShowAll(start);
    
    // 2. Insert Nodes
    InsNode(now, "three", 10, 'M', 3.33); ShowAll(start);
    InsNode(now, "four", 12, 'F', 3.44); ShowAll(start);
    
    // 3. Go Back and Delete
    Goback(&now); // now ถอยจาก two ไป one
    
    // เรียกใช้ DelNode โดยส่ง &start เพื่อป้องกัน error เมื่อลบตัวแรก
    DelNode(&now, &start); ShowAll(start); 
    DelNode(&now, &start); ShowAll(start); 
    DelNode(&now, &start); ShowAll(start);
    
    return 0;
}

void ShowAll(struct studentNode *walk) {
    printf("List: ");
    while (walk != NULL) {
        printf("[%s, %d, %c, %.2f] ", walk->name, walk->age, walk->sex, walk->gpa);
        walk = walk->next;
    }
    printf("\n");
}

struct studentNode *addnode(struct studentNode **start, char *n, int a, char s, float g) {
    struct studentNode *newNode;
    newNode = (struct studentNode *)malloc(sizeof(struct studentNode));
    strcpy(newNode->name, n);
    newNode->age = a;
    newNode->sex = s;
    newNode->gpa = g;
    newNode->next = NULL;
    newNode->back = NULL;

    if (*start == NULL) {
        *start = newNode;
    } else {
        struct studentNode *walk = *start;
        while (walk->next != NULL) {
            walk = walk->next;
        }
        walk->next = newNode;
        newNode->back = walk;
    }
    return newNode;
}

void InsNode(struct studentNode *now, char *n, int a, char s, float g) {
    if (now == NULL) return;
    struct studentNode *newNode = (struct studentNode *)malloc(sizeof(struct studentNode));
    strcpy(newNode->name, n);
    newNode->age = a;
    newNode->sex = s;
    newNode->gpa = g;
    
    // แทรก newNode ต่อจาก now
    newNode->next = now->next;
    newNode->back = now;
    
    if (now->next != NULL) {
        now->next->back = newNode;
    }
    now->next = newNode;
}

void Goback(struct studentNode **now) {
    if (*now != NULL && (*now)->back != NULL) {
        *now = (*now)->back;
    }
}

// แก้ไข: รับ struct studentNode **start เพิ่มเข้ามา
void DelNode(struct studentNode **now, struct studentNode **start) {
    struct studentNode *curr = *now;
    if (curr == NULL) return;

    // 1. ถ้าโหนดที่จะลบคือโหนดแรก (Head) ต้องย้าย start ไปตัวถัดไปก่อน
    if (curr == *start) {
        *start = curr->next;
    }

    // 2. ตัดการเชื่อมโยง (Unlink)
    if (curr->back != NULL) {
        curr->back->next = curr->next;
    } 
    
    if (curr->next != NULL) {
        curr->next->back = curr->back;
    }

    // 3. ย้าย now ไปที่อื่นก่อนลบ เพื่อไม่ให้ pointer ชี้ที่ว่าง
    if (curr->next != NULL) {
        *now = curr->next;
    } else {
        *now = curr->back;
    }

    // 4. คืนหน่วยความจำ
    free(curr);
}
```
